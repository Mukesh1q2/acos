    86→def numpy_safe(obj):
    87→    """Recursively convert numpy types for JSON serialization."""
    88→    if isinstance(obj, np.ndarray):
    89→        return obj.tolist()
    90→    if isinstance(obj, (np.float32, np.float64, np.floating)):
    91→        return float(obj)
    92→    if isinstance(obj, (np.int32, np.int64, np.integer)):
    93→        return int(obj)
    94→    if isinstance(obj, dict):
    95→        return {k: numpy_safe(v) for k, v in obj.items()}
    96→    if isinstance(obj, (list, tuple)):
    97→        return [numpy_safe(v) for v in obj]
    98→    if isinstance(obj, torch.Tensor):
    99→        return numpy_safe(obj.detach().cpu().numpy())
   100→    return obj

   123→def get_fashion_mnist(batch_size=256):
   124→    """
   125→    Fashion-MNIST: 784-dim, 10 classes.
   126→
   127→    Uses sklearn fetch_openml with local caching.
   128→    Returns (train_loader, test_loader, input_dim=784, num_classes=10).
   129→    """
   130→    import pickle
   131→    cache_path = os.path.join(CACHE_DIR, 'fashion_mnist.pkl')
   132→
   133→    if os.path.exists(cache_path):
   134→        with open(cache_path, 'rb') as f:
   135→            data = pickle.load(f)
   136→        X_train, y_train = data['X_train'], data['y_train']
   137→        X_test, y_test = data['X_test'], data['y_test']
   138→    else:
   139→        print("  Downloading Fashion-MNIST via fetch_openml...")
   140→        from sklearn.datasets import fetch_openml
   141→        fashion = fetch_openml('Fashion-MNIST', version=1, as_frame=False)
   142→        X, y = fashion.data, fashion.target.astype(int)
   143→        X_train, X_test = X[:60000], X[60000:]
   144→        y_train, y_test = y[:60000], y[60000:]
   145→        os.makedirs(CACHE_DIR, exist_ok=True)
   146→        with open(cache_path, 'wb') as f:
   147→            pickle.dump({'X_train': X_train, 'y_train': y_train,
   148→                         'X_test': X_test, 'y_test': y_test}, f)
   149→
   150→    X_train = X_train.astype(np.float32) / 255.0
   151→    X_test = X_test.astype(np.float32) / 255.0
   152→
   153→    train_ds = TensorDataset(torch.FloatTensor(X_train), torch.LongTensor(y_train))
   154→    test_ds = TensorDataset(torch.FloatTensor(X_test), torch.LongTensor(y_test))
   155→
   156→    train_loader = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
   157→    test_loader = DataLoader(test_ds, batch_size=batch_size, shuffle=False)
   158→
   159→    return train_loader, test_loader, 784, 10

   474→# ===========================================================================
   475→# 3. TRAINING LOOP (with full metric recording)
   476→# ===========================================================================
   477→
   478→def train_one_epoch(model, loader, optimizer, config_name, beta=0.01,
   479→                    orth_weight=0.0, device='cpu'):
   480→    """
   481→    Train model for one epoch, recording all loss components.
   482→
   483→    Returns dict with epoch-level means of: loss, ce_loss, kl_loss,
   484→    orth_loss, accuracy, active_dims.
   485→    """
   486→    cfg = ABLATION_CONFIGS[config_name]
   487→    model.train()
   488→    metrics = defaultdict(list)
   489→
   490→    for X, y in loader:
   491→        X, y = X.to(device), y.to(device)
   492→        optimizer.zero_grad()
   493→
   494→        if cfg['model_type'] == 'baseline':
   495→            logits, recon, mu, log_var = model(X)
   496→            kl_val = torch.tensor(0.0)
   497→
   498→            if cfg['loss_type'] == 'vae' and beta > 0:
   499→                # β-VAE loss
   500→                ce = F.cross_entropy(logits, y)
   501→                kl_val = -0.5 * torch.sum(1 + log_var - mu.pow(2) - log_var.exp())
   502→                total_loss = ce + beta * kl_val
   503→            else:
   504→                ce = F.cross_entropy(logits, y)
   505→                total_loss = ce
   506→
   507→            orth_loss_val = torch.tensor(0.0)
   508→            latent_for_active = mu  # for active-dim computation
   509→
   510→        else:  # afm
   511→            logits, recon, mu, log_var, kl_val = model(X)
   512→
   513→            ce = F.cross_entropy(logits, y)
   514→            total_loss = ce
   515→
   516→            # Add KL term if configured
   517→            if cfg['loss_type'] in ('rib', 'vae') and beta > 0 and kl_val is not None:
   518→                total_loss = total_loss + beta * kl_val
   519→
   520→            # Orthogonal regularisation for afm_qr config
   521→            orth_loss_val = torch.tensor(0.0)
   522→            if orth_weight > 0:
   523→                with torch.no_grad():
   524→                    S, _ = model.stiefel(mu, log_var)
   525→                S_for_orth = S  # keep graph
   526→                I_K = torch.eye(K_THREADS, device=device)
   527→                gram = torch.bmm(S_for_orth.transpose(1, 2), S_for_orth)  # (B, K, K)
   528→                orth_loss_val = torch.mean((gram - I_K.unsqueeze(0)).pow(2).sum(dim=(-2, -1)))
   529→                total_loss = total_loss + orth_weight * orth_loss_val
   530→
   531→            latent_for_active = mu  # pre-projection latent for active-dim
   532→
   533→        total_loss.backward()
   534→        optimizer.step()
   535→
   536→        # Record metrics
   537→        with torch.no_grad():
   538→            acc = (logits.argmax(1) == y).float().mean().item()
   539→            # Active dimensions: count dims where per-batch variance > threshold
   540→            latent_var = latent_for_active.var(dim=0)
   541→            active_dims = (latent_var > 1e-4).sum().item()
   542→
   543→        metrics['total_loss'].append(total_loss.item())
   544→        metrics['ce_loss'].append(ce.item())
   545→        metrics['kl_loss'].append(kl_val.item() if isinstance(kl_val, torch.Tensor) else float(kl_val))
   546→        metrics['orth_loss'].append(orth_loss_val.item() if isinstance(orth_loss_val, torch.Tensor) else float(orth_loss_val))
   547→        metrics['accuracy'].append(acc)
   548→        metrics['active_dims'].append(active_dims)
   549→
   550→    return {k: float(np.mean(v)) for k, v in metrics.items()}
   551→
   552→
   553→def train_full(model, train_loader, test_loader, config_name,
   554→               epochs=20, lr=1e-3, beta=0.01, orth_weight=0.0,
   555→               device='cpu', verbose=True):
   556→    """
   557→    Full training loop with per-epoch metric recording.
   558→
   559→    Returns dict with:
   560→      - history: list of per-epoch dicts (seed, epoch, all loss components, accuracy, ...)
   561→      - best_test_acc, final_test_acc, total_time
   562→    """
   563→    optimizer = torch.optim.Adam(model.parameters(), lr=lr)
   564→    scheduler = torch.optim.lr_scheduler.StepLR(optimizer, step_size=15, gamma=0.5)
   565→
   566→    history = []
   567→    best_test_acc = 0.0
   568→    t0 = time.time()
   569→
   570→    for epoch in range(epochs):
   571→        ep_start = time.time()
   572→
   573→        # Train
   574→        train_m = train_one_epoch(
   575→            model, train_loader, optimizer, config_name,
   576→            beta=beta, orth_weight=orth_weight, device=device,
   577→        )
   578→
   579→        # Evaluate
   580→        model.eval()
   581→        eval_m = _evaluate_fast(model, test_loader, config_name, device)
   582→
   583→        scheduler.step()
   584→        ep_time = time.time() - ep_start
   585→
   586→        test_acc = eval_m['accuracy']
   587→        if test_acc > best_test_acc:
   588→            best_test_acc = test_acc
   589→
   590→        ep_record = {
   591→            'epoch': epoch,
   592→            'train_total_loss': train_m['total_loss'],
   593→            'train_ce_loss': train_m['ce_loss'],
   594→            'train_kl_loss': train_m['kl_loss'],
   595→            'train_orth_loss': train_m['orth_loss'],
   596→            'train_accuracy': train_m['accuracy'],
   597→            'train_active_dims': train_m['active_dims'],
   598→            'test_accuracy': test_acc,
   599→            'test_loss': eval_m['loss'],
   600→            'epoch_time': ep_time,
   601→        }
   602→        history.append(ep_record)
   603→
   604→        if verbose and (epoch + 1) % 5 == 0:
   605→            print(f"    Epoch {epoch+1}/{epochs}: "
   606→                  f"loss={train_m['total_loss']:.4f} "
   607→                  f"ce={train_m['ce_loss']:.4f} "
   608→                  f"kl={train_m['kl_loss']:.4f} "
   609→                  f"orth={train_m['orth_loss']:.6f} "
   610→                  f"train_acc={train_m['accuracy']:.4f} "
   611→                  f"test_acc={test_acc:.4f} "
   612→                  f"active={train_m['active_dims']:.0f} "
   613→                  f"t={ep_time:.1f}s")
   614→
   615→    total_time = time.time() - t0
   616→
   617→    return {
   618→        'history': history,
   619→        'best_test_acc': best_test_acc,
   620→        'final_test_acc': history[-1]['test_accuracy'] if history else 0,
   621→        'final_test_loss': history[-1]['test_loss'] if history else 0,
   622→        'total_time': total_time,
   623→    }
   624→
   625→
   626→def _evaluate_fast(model, loader, config_name, device='cpu'):
   627→    """Fast evaluation: loss + accuracy only (no latent extraction)."""
   628→    model.eval()
   629→    total_loss = 0.0
   630→    total_correct = 0
   631→    total_samples = 0
   632→
   633→    with torch.no_grad():
   634→        for X, y in loader:
   635→            X, y = X.to(device), y.to(device)
   636→            if ABLATION_CONFIGS[config_name]['model_type'] == 'baseline':
   637→                logits, _, _, _ = model(X)
   638→            else:
   639→                logits, _, _, _, _ = model(X)
   640→            loss = F.cross_entropy(logits, y, reduction='sum')
   641→            total_loss += loss.item()
   642→            total_correct += (logits.argmax(1) == y).sum().item()
   643→            total_samples += y.size(0)
   644→
   645→    return {
   646→        'loss': total_loss / total_samples,
   647→        'accuracy': total_correct / total_samples,
   648→    }

   684→        loader_fn = DATASET_LOADERS[ds_name]

   719→                print(f" best_acc={result['best_test_acc']:.4f} "
   720→                      f"time={result['total_time']:.1f}s")
   721→
   722→            # Summarise across seeds
   723→            accs = [r['best_test_acc'] for r in cfg_runs]
   724→            losses = [r['final_test_loss'] for r in cfg_runs]
   725→            times = [r['total_time'] for r in cfg_runs]
   726→

   728→            final_kl = np.mean([r['history'][-1]['train_kl_loss'] for r in cfg_runs])
   729→            final_active = np.mean([r['history'][-1]['train_active_dims'] for r in cfg_runs])
   730→
   731→            ds_results[cfg_name] = {
   732→                'runs': cfg_runs,
   733→                'summary': {
   734→                    'test_acc_mean': float(np.mean(accs)),
   735→                    'test_acc_std': float(np.std(accs, ddof=1) if len(accs) > 1 else 0.0),
   736→                    'test_loss_mean': float(np.mean(losses)),
   737→                    'test_loss_std': float(np.std(losses, ddof=1) if len(losses) > 1 else 0.0),
   738→                    'final_kl_mean': float(final_kl),
   739→                    'final_active_dims_mean': float(final_active),
   740→                    'time_mean': float(np.mean(times)),
   741→                },
   742→            }
   743→
   744→            print(f"    → acc = {np.mean(accs):.4f} ± {np.std(accs, ddof=1) if len(accs)>1 else 0:.4f}, "
   745→                  f"KL = {final_kl:.4f}, active_dims = {final_active:.0f}")
   746→
   747→        all_results[ds_name] = ds_results
   748→        save_json(f'ablation_{ds_name}', ds_results)

   757→def get_split_mnist(batch_size=256):
   758→    """
   759→    Split-MNIST: Task 1 = digits 0-4, Task 2 = digits 5-9.
   760→
   761→    Returns list of (train_loader, test_loader, input_dim, num_classes) per task.
   762→    For each task, num_classes=5 and labels are remapped to 0-4.
   763→    """
   764→    train_loader, test_loader, _, _ = get_mnist(batch_size=batch_size)
   765→
   766→    tasks = []
   767→    for task_id, digit_range in enumerate([range(0, 5), range(5, 10)]):
   768→        digits = list(digit_range)
   769→
   770→        # Filter train set
   771→        train_X_list, train_y_list = [], []
   772→        for X, y in train_loader:
   773→            mask = sum((y == d) for d in digits).bool()
   774→            train_X_list.append(X[mask])
   775→            train_y_list.append(y[mask] - digits[0])  # remap to 0-4
   776→
   777→        train_X = torch.cat(train_X_list)
   778→        train_y = torch.cat(train_y_list)
   779→        train_ds = TensorDataset(train_X, train_y)
   780→        train_l = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
   781→
   782→        # Filter test set
   783→        test_X_list, test_y_list = [], []
   784→        for X, y in test_loader:
   785→            mask = sum((y == d) for d in digits).bool()
   786→            test_X_list.append(X[mask])
   787→            test_y_list.append(y[mask] - digits[0])
   788→
   789→        test_X = torch.cat(test_X_list)
   790→        test_y = torch.cat(test_y_list)
   791→        test_ds = TensorDataset(test_X, test_y)
   792→        test_l = DataLoader(test_ds, batch_size=batch_size, shuffle=False)
   793→
   794→        tasks.append((train_l, test_l, 784, 5))
   795→
   796→    return tasks

   799→def get_permuted_mnist(batch_size=256, n_tasks=2):
   800→    """
   801→    Permuted-MNIST: Task 1 = original MNIST, Task 2 = fixed permutation of pixels.
   802→
   803→    Returns list of (train_loader, test_loader, input_dim, num_classes) per task.
   804→    """
   805→    train_loader, test_loader, _, _ = get_mnist(batch_size=batch_size)
   806→
   807→    # Fixed permutation (seed=0)
   808→    rng = np.random.RandomState(0)
   809→    perm = rng.permutation(784)
   810→
   811→    tasks = []
   812→    for task_id in range(n_tasks):
   813→        if task_id == 0:
   814→            # Original MNIST
   815→            tasks.append((train_loader, test_loader, 784, 10))
   816→        else:
   817→            # Permuted MNIST
   818→            perm_train_X = []
   819→            for X, y in train_loader:
   820→                perm_train_X.append(X[:, perm])
   821→            perm_train_X = torch.cat(perm_train_X)
   822→            train_y = torch.cat([y for X, y in train_loader])
   823→            train_ds = TensorDataset(perm_train_X, train_y)
   824→            train_l = DataLoader(train_ds, batch_size=batch_size, shuffle=True)
   825→
   826→            perm_test_X = []
   827→            for X, y in test_loader:
   828→                perm_test_X.append(X[:, perm])
   829→            perm_test_X = torch.cat(perm_test_X)
   830→            test_y = torch.cat([y for X, y in test_loader])
   831→            test_ds = TensorDataset(perm_test_X, test_y)
   832→            test_l = DataLoader(test_ds, batch_size=batch_size, shuffle=False)
   833→
   834→            tasks.append((train_l, test_l, 784, 10))
   835→
   836→    return tasks

   839→def run_continual_learning(seeds, epochs_per_task, lr=DEFAULT_LR):
   840→    """
   841→    Run continual-learning benchmarks (Split-MNIST, Permuted-MNIST)
   842→    across all 5 ablation configs.
   843→
   844→    What it measures:
   845→      - Catastrophic forgetting: accuracy drop on earlier tasks after learning new tasks
   846→      - Whether Stiefel projection preserves old knowledge (orthogonal threads)
   847→      - Whether L_RIB reduces forgetting compared to baseline VAE
   848→
   849→    Key hypothesis from v0.1: AFM+L_RIB reduces forgetting from ~22% to ~4%.
   850→    """
   851→    print("\n" + "=" * 70)
   852→    print("CONTINUAL LEARNING EXPERIMENT")
   853→    print("=" * 70)
   854→
   855→    benchmarks = {
   856→        'split_mnist': get_split_mnist,
   857→        'permuted_mnist': get_permuted_mnist,
   858→    }
   859→
   860→    all_results = {}
   861→
   862→    for bench_name, bench_fn in benchmarks.items():
   863→        print(f"\n{'─'*60}")
   864→        print(f"Benchmark: {bench_name}")
   865→        print(f"{'─'*60}")
   866→
   867→        task_data = bench_fn()
   868→        num_tasks = len(task_data)
   869→        task_classes = [nc for _, _, _, nc in task_data]
   870→        input_dim = task_data[0][2]
   871→
   872→        bench_results = {}
   873→
   874→        for cfg_name, cfg in ABLATION_CONFIGS.items():
   875→            print(f"\n  Config: {cfg_name}")
   876→            cfg_runs = []
   877→
   878→            for seed in seeds:
   879→                print(f"    Seed {seed} ...", end='', flush=True)
   880→                set_seed(seed)
   881→
   882→                # Create multi-task model
   883→                if cfg['model_type'] == 'baseline':
   884→                    model = MultiTaskBaseline(
   885→                        input_dim=input_dim, hidden_dim=HIDDEN_DIM,
   886→                        latent_dim=LATENT_DIM, task_classes=task_classes,
   887→                    )
   888→                else:
   889→                    model = MultiTaskAFMLite(
   890→                        input_dim=input_dim, hidden_dim=HIDDEN_DIM,
   891→                        d=D_STIEFEL, K=K_THREADS, task_classes=task_classes,
   892→                    )
   893→
   894→                optimizer = torch.optim.Adam(model.parameters(), lr=lr)
   895→                scheduler = torch.optim.lr_scheduler.StepLR(
   896→                    optimizer, step_size=max(1, epochs_per_task // 2), gamma=0.5)
   897→
   898→                # Accuracy matrix: acc_matrix[i][j] = acc on task j after training task i
   899→                acc_matrix = np.zeros((num_tasks, num_tasks))
   900→
   901→                for task_id in range(num_tasks):
   902→                    train_l, test_l, _, nc = task_data[task_id]
   903→
   904→                    # Train on this task
   905→                    model.train()
   906→                    for ep in range(epochs_per_task):
   907→                        for X, y in train_l:
   908→                            X, y = X.to(DEVICE), y.to(DEVICE)
   909→                            optimizer.zero_grad()
   910→
   911→                            if cfg['model_type'] == 'baseline':
   912→                                logits, mu, log_var = model(X, task_id=task_id)
   913→                                ce = F.cross_entropy(logits, y)
   914→                                kl_val = torch.tensor(0.0)
   915→                                if cfg['loss_type'] == 'vae' and cfg['beta'] > 0:
   916→                                    kl_val = -0.5 * torch.sum(
   917→                                        1 + log_var - mu.pow(2) - log_var.exp())
   918→                                loss = ce + cfg['beta'] * kl_val
   919→                            else:
   920→                                logits, mu, log_var, kl_val = model(X, task_id=task_id)
   921→                                ce = F.cross_entropy(logits, y)
   922→                                kl_val = kl_val if kl_val is not None else torch.tensor(0.0)
   923→                                loss = ce + cfg['beta'] * kl_val
   924→
   925→                                # Orth regularisation for afm_qr
   926→                                if cfg['orth_weight'] > 0:
   927→                                    S, _ = model.stiefel(mu, log_var)
   928→                                    I_K = torch.eye(K_THREADS, device=DEVICE)
   929→                                    gram = torch.bmm(S.transpose(1, 2), S)
   930→                                    orth = torch.mean(
   931→                                        (gram - I_K.unsqueeze(0)).pow(2).sum(dim=(-2, -1)))
   932→                                    loss = loss + cfg['orth_weight'] * orth
   933→
   934→                            loss.backward()
   935→                            optimizer.step()
   936→                        scheduler.step()
   937→
   938→                    # Evaluate on ALL tasks after training this task
   939→                    model.eval()
   940→                    for eval_id in range(num_tasks):
   941→                        _, eval_l, _, _ = task_data[eval_id]
   942→                        correct, total = 0, 0
   943→                        with torch.no_grad():
   944→                            for X, y in eval_l:
   945→                                X, y = X.to(DEVICE), y.to(DEVICE)
   946→                                if cfg['model_type'] == 'baseline':
   947→                                    logits, _, _ = model(X, task_id=eval_id)
   948→                                else:
   949→                                    logits, _, _, _ = model(X, task_id=eval_id)
   950→                                correct += (logits.argmax(1) == y).sum().item()
   951→                                total += y.size(0)
   952→                        acc_matrix[task_id, eval_id] = correct / total
   953→
   954→                # Compute forgetting per task
   955→                forgetting = []
   956→                for t in range(num_tasks - 1):
   957→                    acc_after_learning = acc_matrix[t, t]
   958→                    acc_after_all = acc_matrix[num_tasks - 1, t]
   959→                    forgetting.append(acc_after_learning - acc_after_all)
   960→
   961→                avg_forgetting = float(np.mean(forgetting)) if forgetting else 0.0
   962→
   963→                cfg_runs.append({
   964→                    'seed': seed,
   965→                    'acc_matrix': acc_matrix,
   966→                    'forgetting': forgetting,
   967→                    'avg_forgetting': avg_forgetting,
   968→                })
   969→                print(f" forgetting={avg_forgetting:.4f}")
   970→
   971→            # Summarise
   972→            avg_forgettings = [r['avg_forgetting'] for r in cfg_runs]
   973→            bench_results[cfg_name] = {
   974→                'runs': cfg_runs,
   975→                'summary': {
   976→                    'avg_forgetting_mean': float(np.mean(avg_forgettings)),
   977→                    'avg_forgetting_std': float(
   978→                        np.std(avg_forgettings, ddof=1) if len(avg_forgettings) > 1 else 0.0),
   979→                },
   980→            }
   981→            print(f"    → avg_forgetting = {np.mean(avg_forgettings):.4f} "
   982→                  f"± {np.std(avg_forgettings, ddof=1) if len(avg_forgettings)>1 else 0:.4f}")
   983→
   984→        all_results[bench_name] = bench_results
   985→        save_json(f'continual_{bench_name}', bench_results)
   986→
   987→    return all_results

   994→def run_representation_analysis(seeds, epochs, lr=DEFAULT_LR):
   995→    """
   996→    Analyse learned representations: PCA, t-SNE, silhouette, active dims,
   997→    and representation stability across training.
   998→
   999→    What it measures:
  1000→      - PCA: Do Stiefel representations concentrate variance differently?
  1001→      - t-SNE: Visual cluster structure (2D coords for later plotting)
  1002→      - Silhouette: Quantitative cluster separation by class label
  1003→      - Active dimensions: How many latent dimensions carry signal?
  1004→      - Stability: Do representations drift across training epochs?
  1005→
  1006→    Key hypothesis: AFM representations should be more stable (less drift)
  1007→    because the Stiefel constraint anchors the latent geometry.
  1008→    """
  1009→    print("\n" + "=" * 70)
  1010→    print("REPRESENTATION ANALYSIS")
  1011→    print("=" * 70)

  1196→def compute_pairwise_tests(ablation_results, metric='test_acc'):
  1197→    """
  1198→    Compute pairwise t-tests and Cohen's d between all config pairs,
  1199→    for each dataset.
  1200→
  1201→    What it measures:
  1202→      - Whether differences between configurations are statistically significant
  1203→      - Effect sizes (Cohen's d) to distinguish practical from statistical significance
  1204→
  1205→    Cohen's d interpretation:
  1206→      |d| < 0.2  →  negligible
  1207→      |d| < 0.5  →  small
  1208→      |d| < 0.8  →  medium
  1209→      |d| >= 0.8 →  large
  1210→    """
  1211→    from scipy import stats
  1212→
  1213→    print("\n" + "=" * 70)
  1214→    print("STATISTICAL TESTS: Pairwise t-tests & Cohen's d")
  1215→    print("=" * 70)
  1216→
  1217→    all_stats = {}
  1218→
  1219→    for ds_name, ds_data in ablation_results.items():
  1220→        print(f"\n  Dataset: {ds_name}")
  1221→        ds_stats = {}
  1222→
  1223→        # Collect per-config metric values across seeds
  1224→        config_values = {}
  1225→        for cfg_name, cfg_data in ds_data.items():
  1226→            if metric == 'test_acc':
  1227→                vals = [r['best_test_acc'] for r in cfg_data['runs']]
  1228→            elif metric == 'test_loss':
  1229→                vals = [r['final_test_loss'] for r in cfg_data['runs']]
  1230→            elif metric == 'kl_loss':
  1231→                vals = [r['history'][-1]['train_kl_loss'] for r in cfg_data['runs']]
  1232→            else:
  1233→                vals = [r['best_test_acc'] for r in cfg_data['runs']]
  1234→            config_values[cfg_name] = vals
  1235→
  1236→        config_names = list(config_values.keys())
  1237→
  1238→        for i, c1 in enumerate(config_names):
  1239→            for j, c2 in enumerate(config_names):
  1240→                if i >= j:
  1241→                    continue
  1242→
  1243→                v1 = np.array(config_values[c1])
  1244→                v2 = np.array(config_values[c2])
  1245→
  1246→                # t-test (paired if same seeds, independent otherwise)
  1247→                if len(v1) >= 2 and len(v2) >= 2:
  1248→                    t_stat, p_value = stats.ttest_ind(v1, v2)
  1249→                else:
  1250→                    t_stat, p_value = 0.0, 1.0
  1251→
  1252→                # Cohen's d
  1253→                pooled_std = np.sqrt(
  1254→                    (np.var(v1, ddof=1) + np.var(v2, ddof=1)) / 2
  1255→                ) if len(v1) > 1 and len(v2) > 1 else 1e-10
  1256→                cohens_d = float((np.mean(v1) - np.mean(v2)) / (pooled_std + 1e-10))
  1257→
  1258→                pair_key = f'{c1}_vs_{c2}'
  1259→                ds_stats[pair_key] = {
  1260→                    'config_1': c1,
  1261→                    'config_2': c2,
  1262→                    'mean_1': float(np.mean(v1)),
  1263→                    'mean_2': float(np.mean(v2)),
  1264→                    'diff': float(np.mean(v1) - np.mean(v2)),
  1265→                    't_statistic': float(t_stat),
  1266→                    'p_value': float(p_value),
  1267→                    'cohens_d': cohens_d,
  1268→                    'significant_005': bool(p_value < 0.05),
  1269→                    'effect_size': ('large' if abs(cohens_d) >= 0.8
  1270→                                    else 'medium' if abs(cohens_d) >= 0.5
  1271→                                    else 'small' if abs(cohens_d) >= 0.2
  1272→                                    else 'negligible'),
  1273→                }
  1274→
  1275→                print(f"    {pair_key}: diff={ds_stats[pair_key]['diff']:.4f}, "
  1276→                      f"p={p_value:.4f}, d={cohens_d:.3f} ({ds_stats[pair_key]['effect_size']})")
  1277→
  1278→        all_stats[ds_name] = ds_stats
  1279→
  1280→    save_json('statistical_tests', all_stats)
  1281→    return all_stats
